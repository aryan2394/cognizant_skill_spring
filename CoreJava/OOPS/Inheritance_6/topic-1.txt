"Jo common cheezein hain unhe parent me daal do."

Ab Dekho
class Person {

    String name;
    int age;

}

Student

class Student extends Person {

    void study() {

        System.out.println("Studying");

    }

}

Teacher

class Teacher extends Person {

    void teach() {

        System.out.println("Teaching");

    }

}
