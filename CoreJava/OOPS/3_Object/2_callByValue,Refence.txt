public class Demo {
    public static void main(String args[])
    {
        Random r=new Random(4,5);
        System.out.println("x is "+r.x);
        System.out.println("y is "+r.y);
        fnc(r);
        System.out.println("x is "+r.x);
        System.out.println("y is "+r.y);
    }
    public static void fnc(Random r)
    {
        r.x=r.x+10;
        r.y=r.y+10;
    }
}

class Random{
    int x;
    int y;
    Random(int x,int y)
    {
        this.x=x;
        this.y=y;
    }
}

// output is

// x is 4
// y is 5
// x is 14
// y is 15
