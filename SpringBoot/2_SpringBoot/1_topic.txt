@SpringBootApplication
public class SpringBootDemoApplication {

	public static void main(String[] args) {

        ApplicationContext context=SpringApplication.run(SpringBootDemoApplication.class, args);
        PaymentGateway paymentGateway=context.getBean(PaymentGateway.class);
        paymentGateway.setType("razorPay");
        System.out.println(paymentGateway.getType());
	}

}

@Component
public class PaymentGateway {
    private String type;
    private int retryCount;

    // Getter for type
    public String getType() {
        return type;
    }

    // Setter for type
    public void setType(String type) {
        this.type = type;
    }

    // Getter for retryCount
    public int getRetryCount() {
        return retryCount;
    }

    // Setter for retryCount
    public void setRetryCount(int retryCount) {
        this.retryCount = retryCount;
    }
}

abhi hum main se type ki value set kar rahe hai lekin agar humein chahiye ki hum values ko aise n set karein balki hum apne application.properties se ste karein toh kaise karein
